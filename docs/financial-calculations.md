## Додаток: Калькулятор собівартості

```python
# Калькулятор для швидких розрахунків

def calculate_cost(quantity, with_ai=True):
    # Друк
    if quantity == 1:
        print_cost = 510
    elif quantity <= 5:
        print_cost = 387
    elif quantity <= 10:
        print_cost = 361.5
    else:
        print_cost = 355

    # AI
    ai_cost = 237 if with_ai else 0

    # Payment processing (1.5% від 1299)
    payment_cost = 1299 * 0.015

    # Податки (6.5%)
    tax_cost = (print_cost + ai_cost) * 0.065

    total = print_cost + ai_cost + payment_cost + tax_cost

    return {
        'print': print_cost,
        'ai': ai_cost,
        'payment': payment_cost,
        'tax': tax_cost,
        'total': total
    }

# Приклад
print(calculate_cost(1, with_ai=True))
# {'print': 510, 'ai': 237, 'payment': 19.49, 'tax': 48.56, 'total': 815.05}
```
